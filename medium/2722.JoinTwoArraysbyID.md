# Solution

```jsx
var join = function (arr1, arr2) {
  let mergedArr = [...arr1, ...arr2];
  let result = {};

  for (const item of mergedArr) {
    if (!result[item.id]) {
      result[item.id] = item;
      continue;
    }
    result[item.id] = { ...result[item.id], ...item };
  }
  return Object.values(result);
};
```
