# Solution

```js
/**
 * @param {number} millis
 * @return {Promise}
 */
async function sleep(millis) {
  let t = Date.now();
  return sleep(millis).then(() => {
    console.log(Date.now() - t);
  });
}

/**
 * let t = Date.now()
 * sleep(100).then(() => console.log(Date.now() - t)) // 100
 */
```
