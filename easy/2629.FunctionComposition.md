# Solution

```js
/**
 * @param {Function[]} functions
 * @return {Function}
 */
var compose = function (functions) {
  return function (x) {
    let lastIndex = functions.length - 1;
    let current = x;

    for (let i = lastIndex; i >= 0; i--) {
      current = functions[i](current);
    }

    return current;
  };
};

/**
 * const fn = compose([x => x + 1, x => 2 * x])
 * fn(4) // 9
 */
```
